\section{Benötigte Informationen im System}
\label{sec:Benötigte_Informationen_im_System}

\subsection{Benutzerinformationen}
\label{sec:Benutzerinformationen}

\begin{itemize}
	\item (Lehreinrichtungs-)E-Mail
	\begin{itemize}
		\item Eine E-Mail-Adresse die den dazugehörigen Benutzer und seine Berechtigungen eindeutig identifiziert.
	\end{itemize}
	\item Status im Bezug auf die Lehreinrichtung
	\begin{itemize}
		\item Ergibt sich aus der (Lehreinrichtungs-)E-Mail
	\end{itemize}
	\item Standort des Benutzers
	\begin{itemize}
		\item Der aktuelle Standort des Benutzers von dem aus das System seine Berechnung für einen freien Raum aufstellt.
	\end{itemize}
\end{itemize}

\subsection{Belegungsplan}
\label{sec:Belegungsplan}

\begin{itemize}
	\item wöchentlich wiederkehrender Belegungsplan für jeden Raum
	\begin{itemize}
		\item für jeden Tag/jede Stunde die Belegung feststellen
		\begin{itemize}
			\item Raumnummer/RaumID
			\item Start
			\item Ende
			\item Veranstaltungsname
			\item betreuende Lehrkraft
		\end{itemize}
	\end{itemize}
	\item dynamischer Belegungsplan für jeden Raum
	\begin{itemize}
		\item für jeden Tag/jede Stunde die Belegung, je nach aktueller Auslastung feststellen
		\begin{itemize}
			\item Raumnummer/RaumID
			\item BenutzerID des Buchers
			\item Anzahl Personen im Raum
			\item Start der Buchung
			\item Ende der Buchung
		\end{itemize}
\end{itemize}
